30# 传入过滤之后的数据，通过调用gensim.similarities.Similarity计算余弦相似度 

31def calc_similarity(text1, text2): 

32 texts = [text1, text2] 

33 # 形成词典 

34 dictionary = gensim.corpora.Dictionary(texts) 

35 # 形成向量 

36 corpus = [dictionary.doc2bow(text) for text in texts] 

37 similarity = gensim.similarities.Similarity('-Similarity-index', corpus, num_features=len(dictionary)) 

38 test_corpus_1 = dictionary.doc2bow(text1)
